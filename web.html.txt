<body>
    <div id="search-container">
        <input type="number" id="x-coord" placeholder="X" min="0" max="1000000">
        <input type="number" id="y-coord" placeholder="Y" min="0" max="1000000">
        <button onclick="goToPixel()">Buscar</button>
    </div>
    <div id="container">
        <canvas id="canvas"></canvas>
    </div>
    <script>
        const container = document.getElementById("container");
        const canvas = document.getElementById("canvas");
        const ctx = canvas.getContext("2d");
        let zoomLevel = 1;

        function drawGrid() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = "black";
            ctx.font = `${10 * zoomLevel}px Arial`;
            
            for (let i = 0; i <= 1000000; i += 10000) {
                ctx.fillText(i, i * zoomLevel, 10 * zoomLevel);
                ctx.fillText(i, 10 * zoomLevel, i * zoomLevel);
            }
        }
        drawGrid();

        function goToPixel() {
            let x = parseInt(document.getElementById("x-coord").value);
            let y = parseInt(document.getElementById("y-coord").value);
            if (isNaN(x) || isNaN(y) || x < 0 || x > 1000000 || y < 0 || y > 1000000) {
                alert("Por favor, ingresa coordenadas válidas entre 0 y 1,000,000.");
                return;
            }
            container.scrollLeft = x * zoomLevel;
            container.scrollTop = y * zoomLevel;
        }

        container.addEventListener("wheel", function(event) {
            event.preventDefault();
            let scaleFactor = event.deltaY > 0 ? 0.9 : 1.1;
            zoomLevel *= scaleFactor;
            canvas.style.transform = `scale(${zoomLevel})`;
            drawGrid();
        });
    </script>
